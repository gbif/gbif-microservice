001package org.gbif.jetty;
002
003import java.net.URISyntaxException;
004
005import com.google.common.base.Throwables;
006import org.eclipse.jetty.server.Server;
007import org.eclipse.jetty.server.handler.ContextHandler;
008import org.eclipse.jetty.webapp.WebAppContext;
009
010/**
011 * Utility class to generate the default connectors of an application.
012 */
013public class ContextFactory {
014
015  private static final String WEB_DIR = "webapp";
016  private static final String DESCRIPTOR_PATH = "/WEB-INF/web.xml";
017  public static final String ROOT_CONTEXT = "/";
018  public static final String STOP_CONTEXT = "/stop";
019
020  //The @ is used by Jetty to find registered connectors
021  private static final String VH_HOST_FMT = "@%s";
022
023  /**
024   * Builds an WebAppContext that follows the standard Maven web application.
025   * - The resource base points to  "src/main/webapp/"
026   * - The application descriptor must be in "src/main/webapp/WEB-INF/web.xml"
027   * - The root context is "/"
028   * - The context will be attached to the application connector.
029   */
030  public static ContextHandler buildApplicationContext() {
031    try {
032      //the webapp directory must be loaded from inside the jar file
033      final String resourceBase = ContextFactory.class.getClassLoader().getResource(WEB_DIR).toURI().toString();
034      final WebAppContext root = new WebAppContext();
035      root.setContextPath(ROOT_CONTEXT);
036      root.setDescriptor(resourceBase + DESCRIPTOR_PATH);
037      root.setResourceBase(resourceBase);
038      root.setParentLoaderPriority(true);
039      //set the connector
040      root.setVirtualHosts(new String[] {String.format(VH_HOST_FMT,HttpConnectorFactory.APP_CONNECTOR_NAME)});
041      return root;
042    } catch (URISyntaxException ex){
043      throw Throwables.propagate(ex);
044    }
045  }
046
047  /**
048   * Creates a ContextHandler with a handler to stop the application gracefully.
049   * The context will be attached to the admin connector and accessible at the context "/stop".
050   */
051  public static ContextHandler buildAdminContext(Server server, String secret) {
052    ContextHandler adminContext = new ContextHandler();
053    adminContext.setContextPath(ROOT_CONTEXT);
054    adminContext.setHandler(new StopHandler(server, secret));
055    //set the connector
056    adminContext.setVirtualHosts(new String[] {String.format(VH_HOST_FMT,HttpConnectorFactory.ADMIN_CONNECTOR_NAME)});
057    return adminContext;
058  }
059
060  /**
061   * Default private constructor.
062   * Utility classes don't expose constructors.
063   */
064  private ContextFactory() {
065    //do nothing
066  }
067
068}
