1   package org.gbif.jetty;

3   import java.net.URISyntaxException;
4   
5   import com.google.common.base.Throwables;
6   import org.eclipse.jetty.server.Server;
7   import org.eclipse.jetty.server.handler.ContextHandler;
8   import org.eclipse.jetty.webapp.WebAppContext;
9   
10  /**
11   * Utility class to generate the default connectors of an application.
12   */
13  public class ContextFactory {
14  
15    private static final String WEB_DIR = "webapp";
16    private static final String DESCRIPTOR_PATH = "/WEB-INF/web.xml";
17    public static final String ROOT_CONTEXT = "/";
18    public static final String STOP_CONTEXT = "/stop";
19  
20    //The @ is used by Jetty to find registered connectors
21    private static final String VH_HOST_FMT = "@%s";
22  
23    /**
24     * Builds an WebAppContext that follows the standard Maven web application.
25     * - The resource base points to  "src/main/webapp/"
26     * - The application descriptor must be in "src/main/webapp/WEB-INF/web.xml"
27     * - The root context is "/"
28     * - The context will be attached to the application connector.
29     */
30    public static ContextHandler buildApplicationContext() {
31      try {
32        //the webapp directory must be loaded from inside the jar file
33        final String resourceBase = ContextFactory.class.getClassLoader().getResource(WEB_DIR).toURI().toString();
34        final WebAppContext root = new WebAppContext();
35        root.setContextPath(ROOT_CONTEXT);
36        root.setDescriptor(resourceBase + DESCRIPTOR_PATH);
37        root.setResourceBase(resourceBase);
38        root.setParentLoaderPriority(true);
39        //set the connector
40        root.setVirtualHosts(new String[] {String.format(VH_HOST_FMT,HttpConnectorFactory.APP_CONNECTOR_NAME)});
41        return root;
42      } catch (URISyntaxException ex){
43        throw Throwables.propagate(ex);
44      }
45    }
46  
47    /**
48     * Creates a ContextHandler with a handler to stop the application gracefully.
49     * The context will be attached to the admin connector and accessible at the context "/stop".
50     */
51    public static ContextHandler buildAdminContext(Server server, String secret) {
52      ContextHandler adminContext = new ContextHandler();
53      adminContext.setContextPath(ROOT_CONTEXT);
54      adminContext.setHandler(new StopHandler(server, secret));
55      //set the connector
56      adminContext.setVirtualHosts(new String[] {String.format(VH_HOST_FMT,HttpConnectorFactory.ADMIN_CONNECTOR_NAME)});
57      return adminContext;
58    }
59  
60    /**
61     * Default private constructor.
62     * Utility classes don't expose constructors.
63     */
64    private ContextFactory() {
65      //do nothing
66    }
67  
68  }
