3   import org.gbif.ws.app.ShutdownHolder;
4   
5   import java.io.IOException;
6   import javax.servlet.ServletException;
7   import javax.servlet.http.HttpServletRequest;
8   import javax.servlet.http.HttpServletResponse;
9   
10  import com.google.common.base.Throwables;
11  import org.eclipse.jetty.http.HttpStatus;
12  import org.eclipse.jetty.server.Request;
13  import org.eclipse.jetty.server.Server;
14  import org.eclipse.jetty.server.handler.HandlerWrapper;
15  import org.slf4j.Logger;
16  import org.slf4j.LoggerFactory;
17  
18  /**
19   * Handles the event of stopping the Jetty server.
20   */
21  public class StopHandler extends HandlerWrapper {
22  
23    private static final Logger LOG = LoggerFactory.getLogger(StopHandler.class);
24    private static final String SECRET_PARAM = "secret";
25  
26    //Jetty server to be stopped
27    private final Server server;
28  
29    //Secret to stop the server
30    private final String secret;
31  
32    /**
33     * Creates an instance of the StopHandler for the 'server' parameter using the stop 'secret'.
34     */
35    public StopHandler(Server server, String secret) {
36      this.server = server;
37      this.secret = secret;
38    }
39  
40    /**
41     * Initiates the stop process.
42     * Any unforeseen error returns a INTERNAL_SERVER_ERROR.
43     */
44    @Override
45    public void handle(
46      String target, Request baseRequest, HttpServletRequest request, HttpServletResponse response
47    ) throws IOException, ServletException {
48      try {
49        if (target.equals(ContextFactory.STOP_CONTEXT)) {
50          stopServer(request, response);
51        } else {
52          super.handle(target,baseRequest,request,response);
53        }
54      } catch (Exception ex) {
55        response.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
56      }
57    }
58  
59    /**
60     * Stops the server using an HttpRequest.
61     * From the request object the "secret" parameters is obtained, if it matches against the secret the server is
62     * stopped.
63     * If the secret parameter is null or invalid, and HttpStatus.FORBIDDEN_403 is returned in the HttpResponse.
64     * The stop process is performed in separate thread to avoid race conditions with the actual Http call.
65     */
66    private void stopServer(HttpServletRequest request, HttpServletResponse response) throws IOException {
67      //Validates the secret
68      if (request.getParameterMap().containsKey(SECRET_PARAM) && secret.equals(request.getParameter(SECRET_PARAM))) {
69        LOG.info("Stopping Jetty");
70        response.setStatus(HttpStatus.ACCEPTED_202);
71        response.flushBuffer();
72        try { // Stops the server in a new thread to guarantee the execution of the current response
73          new Thread() {
74            @Override
75            public void run() {
76              ShutdownHolder.stopServer(server);
77              System.exit(0);
78            }
79          }.start();
80        } catch (Exception ex) {
81          LOG.error("Unable to stop Jetty", ex);
82          throw Throwables.propagate(ex);
83        }
84      } else { //invalid secret parameter
85        response.sendError(HttpServletResponse.SC_UNAUTHORIZED);
86      }
87    }
88  }
