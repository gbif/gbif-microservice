001package org.gbif.jetty;
002
003import org.gbif.ws.discovery.conf.ServiceConfiguration;
004import org.gbif.ws.discovery.lifecycle.DiscoveryLifeCycle;
005
006import org.eclipse.jetty.server.Connector;
007import org.eclipse.jetty.server.Handler;
008import org.eclipse.jetty.server.Server;
009import org.eclipse.jetty.server.handler.ContextHandlerCollection;
010
011/**
012 * Utility class to generate instances of Jetty Servers.
013 * The default values for the instance are:
014 * -  gracefulShutdown period: 1 second.
015 * - stopAtShutdown: true.
016 */
017public class ServerFactory {
018
019  //Graceful shutdown timeout
020  private int gracefulShutdown = 1000; //1 second
021
022  //Shutdown the server when it's stopped
023  private boolean stopAtShutdown = true;
024
025  public int getGracefulShutdown() {
026    return gracefulShutdown;
027  }
028
029  public void setGracefulShutdown(int gracefulShutdown) {
030    this.gracefulShutdown = gracefulShutdown;
031  }
032
033  public boolean isStopAtShutdown() {
034    return stopAtShutdown;
035  }
036
037  public void setStopAtShutdown(boolean stopAtShutdown) {
038    this.stopAtShutdown = stopAtShutdown;
039  }
040
041  /**
042   * Builds a Jetty Server instance using the configuration class.
043   * From the configuration class the following fields are used:
044   * - stopSecret: stop password used by the StopHandler.
045   * - Http connectors definition: application and admin
046   * - Contexts: default web application and stop context.
047   * If the configuration instance isDiscoverable registers a DiscoveryLifeCycle that handles the discovery process.
048   */
049  public Server build(ServiceConfiguration configuration) {
050    Server server = new Server();
051    server.setConnectors(buildConnectors(configuration,server));
052    server.setStopTimeout(gracefulShutdown);
053    server.setHandler(buildContexts(server, configuration.getStopSecret()));
054    if (configuration.isDiscoverable()) { //Register the discovery lifecycle
055      server.addLifeCycleListener(new DiscoveryLifeCycle(configuration));
056    }
057    return server;
058  }
059
060  /**
061   * Builds the admin and application connectors.
062   */
063  private static Connector[] buildConnectors(ServiceConfiguration configuration, Server server) {
064    final HttpConnectorFactory appConnectorFactory = HttpConnectorFactory.application(server);
065    appConnectorFactory.setPort(configuration.getHttpPort());
066    final HttpConnectorFactory adminConnectorFactory = HttpConnectorFactory.admin(server);
067    adminConnectorFactory.setPort(configuration.getHttpAdminPort());
068    return new Connector[] {appConnectorFactory.build(), adminConnectorFactory.build()};
069  }
070
071  /**
072   * Builds the web application and stop contexts.
073   */
074  private static ContextHandlerCollection buildContexts(Server server, String secret) {
075    ContextHandlerCollection contexts = new ContextHandlerCollection();
076    contexts.setHandlers(new Handler[] {ContextFactory.buildApplicationContext(),
077      ContextFactory.buildAdminContext(server, secret)});
078    return contexts;
079  }
080}
