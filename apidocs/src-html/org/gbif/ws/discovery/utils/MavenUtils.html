001package org.gbif.ws.discovery.utils;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.util.Enumeration;
006import java.util.jar.JarEntry;
007import java.util.jar.JarFile;
008
009import com.google.common.base.Throwables;
010import org.apache.maven.model.Model;
011import org.apache.maven.model.io.xpp3.MavenXpp3Reader;
012import org.apache.maven.project.MavenProject;
013import org.codehaus.plexus.util.xml.pull.XmlPullParserException;
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017/**
018 * Utility class to extract information from Maven projects.
019 */
020public class MavenUtils {
021
022  private static final String POM = "pom.xml";
023  private static final String ERROR_MSG = "Error reading the maven project";
024
025  private static final Logger LOG = LoggerFactory.getLogger(MavenUtils.class);
026
027  /**
028   * Reads the pom.xml file and returns a MavenProjects instance.
029   */
030  public static MavenProject getMavenProject() throws IOException {
031    try(InputStream pomFile = (MavenUtils.class.getResourceAsStream(getPOMPath()))) {
032      MavenXpp3Reader mavenReader = new MavenXpp3Reader();
033      Model model = mavenReader.read(pomFile);
034      return new MavenProject(model);
035    } catch (IOException|XmlPullParserException e) {
036      LOG.error(ERROR_MSG, e);
037      throw Throwables.propagate(e);
038    }
039  }
040
041  /**
042   * Gets the path to the pom.xml file that is inside the jar file.
043   */
044  private static String getPOMPath() throws IOException {
045    String jarPath = MavenUtils.class.getResource("").getPath().replaceAll("file:", "");
046    jarPath = jarPath.replaceAll("!/" + MavenUtils.class.getPackage().getName().replaceAll("\\.", "/") + '/', "");
047    try (JarFile jarFile = new JarFile(jarPath)) {
048      Enumeration<JarEntry> entries = jarFile.entries();
049      while (entries.hasMoreElements()) {
050        JarEntry jarEntry = entries.nextElement();
051        if (jarEntry.getName().endsWith(POM)) {
052          return '/' + jarEntry.getName();
053        }
054      }
055    }
056    throw new IllegalStateException("Error reading pom.xml file");
057  }
058
059  /**
060   * Private constructor.
061   * Utility classes must have a private constructor.
062   */
063  private MavenUtils() {
064    // empty private constructor
065  }
066}
