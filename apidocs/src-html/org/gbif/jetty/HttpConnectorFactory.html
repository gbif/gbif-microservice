001package org.gbif.jetty;
002
003import org.eclipse.jetty.server.Connector;
004import org.eclipse.jetty.server.Server;
005import org.eclipse.jetty.server.ServerConnector;
006
007/**
008 * Factory class that provides instances of the Connectors used by an application.
009 * Contains factory methods for the default connectors: application and admin.
010 */
011public class HttpConnectorFactory {
012
013  public static final String APP_CONNECTOR_NAME = "application";
014  public static final String ADMIN_CONNECTOR_NAME = "admin";
015  private int maxRequestHeaderSize = 8192; //8 Kilobytes
016  private int idleTimeout = 30000; //30 seconds
017  //Http port
018  private int port = 8080;  //default http port
019  //Connector name
020  private String name;
021
022  //Jetty server
023  private Server server;
024
025  /**
026   * Creates an instance of the admin connector.
027   * The default values for this instance are:
028   * - name:  admin
029   * - port: 8081
030   * - threadPool: QueuedThreadPool(maxThreadPoolSize)
031   */
032  public static HttpConnectorFactory admin(Server server) {
033    final HttpConnectorFactory httpConnectorFactory = new HttpConnectorFactory();
034    httpConnectorFactory.port = 8081;  //default admin port
035    httpConnectorFactory.name = ADMIN_CONNECTOR_NAME;
036    httpConnectorFactory.server = server;
037    return httpConnectorFactory;
038  }
039
040  /**
041   * Creates an instance of the application connector.
042   * The instance returned is a SelectorChannelConnector which by default has a threadPool implementation.
043   * The default values for this instance are:
044   * - name:  application
045   * - port: 8080
046   * - threadPool: none
047   */
048  public static HttpConnectorFactory application(Server server) {
049    final HttpConnectorFactory httpConnectorFactory = new HttpConnectorFactory();
050    httpConnectorFactory.name = APP_CONNECTOR_NAME;
051    httpConnectorFactory.server = server;
052    return httpConnectorFactory;
053  }
054
055  public int getMaxRequestHeaderSize() {
056    return maxRequestHeaderSize;
057  }
058
059  public void setMaxRequestHeaderSize(int maxRequestHeaderSize) {
060    this.maxRequestHeaderSize = maxRequestHeaderSize;
061  }
062
063  public int getIdleTimeout() {
064    return idleTimeout;
065  }
066
067  public void setIdleTimeout(int idleTimeout) {
068    this.idleTimeout = idleTimeout;
069  }
070
071  public int getPort() {
072    return port;
073  }
074
075  public void setPort(int port) {
076    this.port = port;
077  }
078
079  public String getName() {
080    return name;
081  }
082
083  public void setName(String name) {
084    this.name = name;
085  }
086
087  public Server getServer() {
088    return server;
089  }
090
091  public void setServer(Server server) {
092    this.server = server;
093  }
094
095  /**
096   * Builds an instance of a Connector using the values provider by the HttpConnectorFactory instance.
097   */
098  public Connector build() {
099    final ServerConnector httpConnector = new ServerConnector(server);
100    httpConnector.setPort(port);
101    httpConnector.setIdleTimeout(idleTimeout);
102    httpConnector.setName(name);
103    return httpConnector;
104  }
105
106}
