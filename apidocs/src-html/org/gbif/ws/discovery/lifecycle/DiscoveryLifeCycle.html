001package org.gbif.ws.discovery.lifecycle;
002
003import org.gbif.ws.discovery.conf.ServiceConfiguration;
004import org.gbif.ws.discovery.conf.ServiceDetails;
005import org.gbif.ws.discovery.conf.ServiceStatus;
006import org.gbif.ws.discovery.utils.MavenUtils;
007
008import java.io.IOException;
009
010import com.google.common.base.Throwables;
011import com.google.common.io.Closer;
012import org.apache.curator.framework.CuratorFramework;
013import org.apache.curator.framework.CuratorFrameworkFactory;
014import org.apache.curator.retry.ExponentialBackoffRetry;
015import org.apache.curator.x.discovery.ServiceDiscovery;
016import org.apache.curator.x.discovery.ServiceDiscoveryBuilder;
017import org.apache.curator.x.discovery.ServiceInstance;
018import org.apache.curator.x.discovery.UriSpec;
019import org.apache.curator.x.discovery.details.JsonInstanceSerializer;
020import org.apache.maven.project.MavenProject;
021import org.eclipse.jetty.util.component.LifeCycle;
022import org.slf4j.Logger;
023import org.slf4j.LoggerFactory;
024
025/**
026 * Listener that handles the registration process of the executing application into the Zookeeper discoveryService service.
027 */
028public class DiscoveryLifeCycle implements LifeCycle.Listener {
029
030  private static final Logger LOG = LoggerFactory.getLogger(DiscoveryLifeCycle.class);
031  private final ServiceConfiguration configuration;
032  private ServiceDiscovery<ServiceDetails> discoveryService;
033  private CuratorFramework curatorClient;
034  private ServiceInstance<ServiceDetails> serviceInstance;
035  // Keeps references to the closable elements: curatorClient and discoveryService.
036  private final Closer closer = Closer.create();
037
038  /**
039   * Creates an instance using the fields zkPath and zkHost of the configuration class.
040   */
041  public DiscoveryLifeCycle(ServiceConfiguration configuration) {
042    this.configuration = configuration;
043  }
044
045  /**
046   * Required services, curator client and discoveryService services, are instantiated while  the application is starting.
047   */
048  @Override
049  public void lifeCycleStarting(LifeCycle event) {
050    curatorClient = curator(configuration);
051    LOG.info("Curator client started");
052    discoveryService = discovery();
053    LOG.info("Discovery service created");
054    LOG.info("Registering service");
055    registerService(configuration);
056    LOG.info("Service registered {}", serviceInstance.toString());
057  }
058
059  /**
060   * Once the application is started the service is registered in the discoveryService services.
061   */
062  @Override
063  public void lifeCycleStarted(LifeCycle event) {
064    updateServiceStatus(ServiceStatus.RUNNING);
065    LOG.info("Service registered");
066  }
067
068  /**
069   * In case of application failure the service is unregistered.
070   */
071  @Override
072  public void lifeCycleFailure(LifeCycle event, Throwable cause) {
073    updateServiceStatus(ServiceStatus.FAILED);
074    unRegisterService();
075  }
076
077  /**
078   * While the application is stopping the service is unregistered.
079   */
080  @Override
081  public void lifeCycleStopping(LifeCycle event) {
082    updateServiceStatus(ServiceStatus.STOPPING);
083  }
084
085  /**
086   * Once the application is stopped: do nothing.
087   * The service must be unregistered in the event lifeCycleStopping.
088   */
089  @Override
090  public void lifeCycleStopped(LifeCycle event) {
091    updateServiceStatus(ServiceStatus.STOPPED);
092    unRegisterService();
093    LOG.info("Discovery services have been stopped");
094  }
095
096  public void unRegisterService() {
097    try {
098      if (discoveryService != null && serviceInstance != null) {
099        discoveryService.unregisterService(serviceInstance);
100        LOG.info("Service instance has been unregistered");
101      }
102      closer.close();
103      LOG.info("All the resources haven been closed");
104    } catch (Exception ex) {
105      LOG.error("Error unregistering services", ex);
106      throw Throwables.propagate(ex);
107    }
108  }
109
110  /**
111   * Registers a new service instance in Zookeeper using the ServiceDiscovery.
112   */
113  public void registerService(ServiceConfiguration configuration) {
114    try {
115      serviceInstance = registerServiceInstance(configuration, ServiceStatus.STARTING);
116      discoveryService.registerService(serviceInstance);
117    } catch (Exception e) {
118      LOG.error("Error registering the service", e);
119      throw Throwables.propagate(e);
120    }
121  }
122
123  /**
124   * Updates the service instance status in Zookeeper.
125   */
126  public void updateServiceStatus(ServiceStatus serviceStatus) {
127    try {
128      if(serviceInstance != null) {
129        serviceInstance.getPayload().setStatus(serviceStatus);
130        discoveryService.updateService(serviceInstance);
131      }
132    } catch (Exception e) {
133      LOG.error("Error updating service status", e);
134      throw Throwables.propagate(e);
135    }
136  }
137
138  /**
139   * Builds a new instance of a ServiceDetails class.
140   * Populates the artifact attributes using the Maven pom.xml file.
141   */
142  public static ServiceDetails serviceDetails(ServiceConfiguration configuration) throws IOException {
143    MavenProject mavenProject = MavenUtils.getMavenProject();
144    ServiceDetails serviceDetails = new ServiceDetails();
145    serviceDetails.setServiceConfiguration(configuration);
146    serviceDetails.setArtifactId(mavenProject.getArtifactId());
147    serviceDetails.setGroupId(mavenProject.getGroupId());
148    serviceDetails.setVersion(mavenProject.getVersion());
149    return serviceDetails;
150  }
151
152  /**
153   * Builds a new instance of a CuratorFramework client.
154   */
155  public CuratorFramework curator(ServiceConfiguration configuration) {
156    CuratorFramework curator = CuratorFrameworkFactory.builder()
157      .connectString(configuration.getZkHost())
158      .namespace(configuration.getZkPath())
159      .retryPolicy(new ExponentialBackoffRetry(1000, 3))
160      .build();
161    curator.start();
162    return closer.register(curator);
163  }
164
165  /**
166   * Builds a new instance of a ServiceDiscovery.
167   */
168  protected ServiceDiscovery<ServiceDetails> discovery() {
169    JsonInstanceSerializer<ServiceDetails> serializer =
170      new JsonInstanceSerializer<ServiceDetails>(ServiceDetails.class);
171    return closer.register(ServiceDiscoveryBuilder.builder(ServiceDetails.class)
172                             .client(curatorClient)
173                             .basePath("/")
174                             .serializer(serializer)
175                             .build());
176  }
177
178  /**
179   * Registers a new instance of a ServiceInstance in a specific status.
180   */
181  private static ServiceInstance<ServiceDetails> registerServiceInstance(ServiceConfiguration configuration, ServiceStatus serviceStatus) {
182    try {
183      ServiceDetails serviceDetails = serviceDetails(configuration);
184      serviceDetails.setStatus(serviceStatus);
185      return ServiceInstance.<ServiceDetails>builder()
186        .name(serviceDetails.getName())
187        .payload(serviceDetails)
188        .port(configuration.getHttpPort())
189        .uriSpec(new UriSpec(serviceDetails.getExternalUrl()))
190        .build();
191    } catch (Exception e) {
192      LOG.error("Error creating a service instance", e);
193      throw Throwables.propagate(e);
194    }
195  }
196}
